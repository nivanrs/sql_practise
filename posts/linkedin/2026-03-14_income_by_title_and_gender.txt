𝗦𝗤𝗟 𝗦𝗼𝗹𝘂𝘁𝗶𝗼𝗻
─────────────────────────────
WITH cte_total_bonus AS (
    SELECT
        worker_ref_id,
        SUM(bonus) AS total_bonus
    FROM sf_bonus
    GROUP BY worker_ref_id
)
SELECT
    e.employee_title,
    e.sex,
    AVG(e.salary + b.total_bonus) AS average_compensation
FROM sf_employee e
INNER JOIN cte_total_bonus b
    ON e.id = b.worker_ref_id
GROUP BY 1, 2;
